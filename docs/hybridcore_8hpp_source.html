HArD::Core3D
Hybrid Arbitrary Degree::Core 3D - Library to implement 3D schemes with edge and cell polynomials as unknowns
  • src

hybridcore.hpp
1 // Core data structures and methods required to implement hybrid schemes in 3D (polynomial unknowns
2 // in the cells and on the faces, such as Hybrid High-order (HHO) schemes).
3 //
4 // Provides:
5 // - Hybrid polynomial basis functions (on the cells and faces of the mesh)
6 // - Generic routines to create quadrature nodes over cells and faces of the mesh
7 // - Interpolation of general functions onto the HHO space
8 // - Methods for integrating, evaluating, and computing norms of HHO solutions
9 //
10 // Author: Jerome Droniou (jerome.droniou@monash.edu)
11 //
12 
13 /*
14  *
15  * This library was developed around HHO methods, although some parts of it have a more
16  * general purpose. If you use this code or part of it in a scientific publication,
17  * please mention the following book as a reference for the underlying principles
18  * of HHO schemes:
19  *

304  template <typename Function>
305  void HybridCore::quadrature_over_cell(const size_t iT, const Function& f) const {
306  assert(iT < _mesh_ptr->n_cells());
307  QuadratureRule quadT = generate_quadrature_rule(*_mesh_ptr->cell(iT), 2 * _Ldeg + 2);
308  for (size_t iqn = 0; iqn < quadT.size(); iqn++) {
309  f(iqn, quadT[iqn].x, quadT[iqn].y, quadT[iqn].z, quadT[iqn].w);
310  }
311  }
312 
313  template <typename Function>
314  void HybridCore::quadrature_over_face(const size_t iF, const Function& f) const {
315  assert(iF < _mesh_ptr->n_faces());
316 
317  QuadratureRule quadF = generate_quadrature_rule(*_mesh_ptr->face(iF), 2 * _K + 2);
318  for (size_t iqn = 0; iqn < quadF.size(); iqn++) {
319  f(iqn, quadF[iqn].x, quadF[iqn].y, quadF[iqn].z, quadF[iqn].w);
320  }
321  }

323  template <typename Function>
324  double HybridCore::integrate_over_cell(const size_t iT, const Function& f) const {
325  assert(iT < _mesh_ptr->n_cells());
326  double ans = 0.0;
327  quadrature_over_cell(iT, [&ans, &f](size_t iQN, double x, double y, double z,
328  double w) { ans += w * f(x, y, z); });
329  return ans;
330  }

357  template<typename Function>
358  Eigen::VectorXd HybridCore::interpolate(const Function& f, size_t doe) const {
359  Eigen::VectorXd XTF = Eigen::VectorXd::Zero(_ntotal_dofs);
360 
361  // Face projections
362  for (size_t iF = 0; iF < _mesh_ptr->n_faces(); iF++) {
363  Face* face = _mesh_ptr->face(iF);
364 
365  // Mass matrix on face
366  QuadratureRule quadF = generate_quadrature_rule(*face, doe);
367  std::vector<Eigen::ArrayXd> phiF_quadF = basis_quad("face", iF, quadF, _K);
368  Eigen::MatrixXd MF = gram_matrix(phiF_quadF, phiF_quadF, _nlocal_face_dofs, _nlocal_face_dofs, quadF, true);
369 
370  // Vector of integral of f against face basis functions
371  Eigen::VectorXd bF = Eigen::VectorXd::Zero(_nlocal_face_dofs);
372  for (size_t i = 0; i < _nlocal_face_dofs; i++) {
373  for (size_t iqn = 0; iqn < quadF.size(); iqn++){
374  bF(i) += quadF[iqn].w * phiF_quadF[i](iqn) * f(quadF[iqn].x, quadF[iqn].y, quadF[iqn].z);
375  }
376  }
377 
378  // Vector of coefficients (on cell basis functions) of the L2(F) projection of f
379  Eigen::VectorXd UF = MF.ldlt().solve(bF);
380 
381  // Fill in he complete vector of DOFs
382  XTF.segment(_ntotal_cell_dofs + iF * _nlocal_face_dofs, _nlocal_face_dofs) = UF;
383  }
384 
385  for (size_t iT = 0; iT < _mesh_ptr->n_cells(); iT++) {
386  Cell* cell = _mesh_ptr->cell(iT);
387  size_t nfaces = cell->n_faces();
388 
389  // Mass matrix in cell
390  QuadratureRule quadT = generate_quadrature_rule(*cell, doe);
391  std::vector<Eigen::ArrayXd> phiT_quadT = basis_quad("cell", iT, quadT, _Ldeg);
392  Eigen::MatrixXd MT = gram_matrix(phiT_quadT, phiT_quadT, _nlocal_cell_dofs, _nlocal_cell_dofs, quadT, true);
393 
394  // Vector of integrals of f against basis functions
395  Eigen::VectorXd bT = Eigen::VectorXd::Zero(_nlocal_cell_dofs);
396  for (size_t i = 0; i < _nlocal_cell_dofs; i++) {
397  for (size_t iqn = 0; iqn < quadT.size(); iqn++){
398  bT(i) += quadT[iqn].w * phiT_quadT[i](iqn) * f(quadT[iqn].x, quadT[iqn].y, quadT[iqn].z);
399  }
400  }
401 
402  // Vector of coefficients (on cell basis functions) of the L2(T) projection of f
403  Eigen::VectorXd UT = MT.ldlt().solve(bT);
404 
405  // Fill in the complete vector of DOFs
406  size_t offset_T = iT * _nlocal_cell_dofs;
407  XTF.segment(offset_T, _nlocal_cell_dofs) = UT;
408 
409  // Special case of L=-1, we replace the previously computed cell value with the proper average of face values
410  if ( _L==-1 ){
411  // Weights corresponding to the values in cells/on faces
412  Eigen::VectorXd barycoefT = compute_weights(iT);
413  // We transform these weights so that they apply to the coefficients on the basis functions
414  Eigen::Vector3d xT = cell->center_mass();
415  double phiT_cst = cell_basis(iT,0)(xT.x(), xT.y(), xT.z());
416  for (size_t ilF = 0; ilF < nfaces; ilF++){
417  Eigen::Vector3d xF = cell->face(ilF)->center_mass();
418  size_t iF = cell->face(ilF)->global_index();
419  double phiF_cst = face_basis(iF,0)(xF.x(), xF.y(), xF.z());
420  barycoefT(ilF) *= phiF_cst / phiT_cst;
421  }
422 
423  XTF(iT) = 0;
424  for (size_t ilF = 0; ilF < nfaces; ilF++) {
425  size_t iF = cell->face(ilF)->global_index();
426  XTF(iT) += barycoefT(ilF) * XTF(_ntotal_cell_dofs + iF);
427  }
428  }
429 
430  }
431 
432  return XTF;
433  }
