64  template<>
65  struct PolynomialSpaceDimension<Cell>
66  {
68  static size_t Poly(int k)
69  {
70  return (k >= 0 ? (k * (k + 1) * (2 * k + 1) + 9 * k * (k + 1) + 12 * (k + 1)) / 12 : 0);
71  }
73  static size_t Goly(int k)
74  {
75  return (k >= 0 ? PolynomialSpaceDimension<Cell>::Poly(k + 1) - 1 : 0);
76  }
78  static size_t GolyOrth(int k)
79  {
80  return 3 * PolynomialSpaceDimension<Cell>::Poly(k) - PolynomialSpaceDimension<Cell>::Goly(k);
81  }
83  static size_t Roly(int k)
84  {
85  return (k >= 0 ? PolynomialSpaceDimension<Cell>::GolyOrth(k + 1) : 0);
86  }
88  static size_t RolyOrth(int k)
89  {
90  return 3 * PolynomialSpaceDimension<Cell>::Poly(k) - PolynomialSpaceDimension<Cell>::Roly(k);
91  }
92  };

95  struct PolynomialSpaceDimension<Face>
96  {
98  static size_t Poly(int k)
99  {
100  return (k >= 0 ? (k + 1) * (k + 2) / 2 : 0);

103  static size_t Roly(int k)
104  {
105  return (k >= 0 ? PolynomialSpaceDimension<Face>::Poly(k + 1) - 1 : 0);
106  }
