56  struct PolynomialSpaceDimension<Cell>
57  {
59  static size_t Poly(int k)
60  {
61  return (k >= 0 ? (k * (k + 1) * (2 * k + 1) + 9 * k * (k + 1) + 12 * (k + 1)) / 12 : 0);
62  }
64  static size_t Goly(int k)
65  {
66  return (k >= 0 ? PolynomialSpaceDimension<Cell>::Poly(k + 1) - 1 : 0);
67  }
69  static size_t GolyCompl(int k)
70  {
71  return 3 * PolynomialSpaceDimension<Cell>::Poly(k) - PolynomialSpaceDimension<Cell>::Goly(k);
72  }
74  static size_t Roly(int k)
75  {
76  return (k >= 0 ? PolynomialSpaceDimension<Cell>::GolyCompl(k + 1) : 0);
